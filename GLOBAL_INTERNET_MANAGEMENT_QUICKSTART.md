```python
from internet_orchestrator import GlobalInternetOrchestrator

orchestrator = GlobalInternetOrchestrator()

# Connect any internet service

    name="AWS",
    service_type=InternetServiceType.CLOUD_PLATFORM,
    endpoint="https://api.aws.com",
    api_key="key_aws",
    capabilities=["deployment", "scaling", "monitoring"]
)

# Grant auto-deploy rights

    service.service_id,
    auto_deploy=True,
    income_share=15.0

    service.service_id,
    version="2.0",

)

    service.service_id,

)
```

### 3. **Permission & Rights Management**
```python
from global_rights_manager import GlobalRightsManager

manager = GlobalRightsManager()

# Grant a complete role
rights = manager.grant_role(
    entity_id="operator_01",
    entity_type="user",
    entity_name="Operations Team",
    role_id="operator",  # Has 6 rights: deploy, auto-deploy, etc.
    expires_in_days=180
)

# Check permission
can_deploy, reason = manager.can_perform_action(
    entity_id="operator_01",
    right_type=RightType.AUTO_DEPLOY
)
# Returns: (True, "Authorized with access level 9/10")

# Create partnership agreement
agreement = manager.create_partnership_agreement(
    partner_id="partner_001",

    revenue_share=25.0,
    term_months=24
)

# Generate API key with rate limiting

    entity_id="dev_01",
```
